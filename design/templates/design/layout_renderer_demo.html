{% trans 'Integration Examples' %}

{% trans 'Brand Builder Integration' %}

from design.layout_renderer import LayoutRenderer

# In BrandingPreviewFrameView
renderer = LayoutRenderer(
    page_type='home',
    tier='C',
    context={
        'branding': branding,
        'preview_mode': True,
        'isolation_type': 'brand_builder',  # → hf-content-preview
    }
)
html = renderer.render()
css = branding.generate_css()

# Template receives html_content and css_content
# CSS isolation is already applied by LayoutRenderer

{% trans 'Page Editor Integration' %}

from design.layout_renderer import LayoutRenderer

# In page editor preview view
renderer = LayoutRenderer(
    page_type=page.page_type,
    tier=get_page_tier(page),
    context={
        'page': page,
        'preview_mode': True,
        'isolation_type': 'page_editor',  # → pb-content-preview
    }
)
html = renderer.render()

# Template receives html_content
# CSS isolation is already applied by LayoutRenderer
