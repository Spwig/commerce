Usage Instructions

1. Include Required Files

{% templatetag openblock %} load static {% templatetag closeblock %}

<!-- CSS -->
<link rel="stylesheet" href="{% templatetag openblock %} static 'catalog/css/attribute_selector.css' {% templatetag closeblock %}">

<!-- JavaScript -->
<script src="{% templatetag openblock %} static 'catalog/js/attribute_selector.js' {% templatetag closeblock %}"></script>

2. Include the Widget Template

{% templatetag openblock %} include 'catalog/partials/attribute_selector.html' with attributes=product.available_attributes {% templatetag closeblock %}

3. Pass Variant Data to JavaScript

<div class="attribute-selector-widget"
     data-product-id="{{ product.id }}"
     data-product-type="{{ product.product_type }}"
     data-variants='{{ product.variants|safe }}'>
    ...
</div>

4. Listen to Events (Optional)

// Listen for variant changes
document.querySelector('.attribute-selector-widget')
    .addEventListener('variantChange', function(e) {
        const variant = e.detail.variant;
        console.log('Selected variant:', variant);

        // Update price, image, etc.
        updateProductDisplay(variant);
    });

// Listen for attribute changes
document.querySelector('.attribute-selector-widget')
    .addEventListener('attributeChange', function(e) {
        console.log('Attribute changed:', e.detail);
    });

5. API Views Setup

The widget expects product data from the API in this format: 

{
    "id": 1,
    "name": "Product Name",
    "product_type": "variable",
    "available_attributes": [
        {
            "id": 1,
            "name": "Color",
            "slug": "color",
            "type": "color",
            "is_required": true,
            "values": [
                {
                    "id": 1,
                    "value": "Red",
                    "slug": "red",
                    "color_hex": "#FF0000"
                }
            ]
        }
    ],
    "variants": [
        {
            "id": 1,
            "name": "Red - Small",
            "sku": "PROD-001-RED-SM",
            "attributes_structured": [
                {"id": 1, "value": "Red"},
                {"id": 3, "value": "Small"}
            ],
            "effective_price": {
                "amount": "99.99",
                "currency": "USD"
            },
            "stock_quantity": 10
        }
    ]
}
