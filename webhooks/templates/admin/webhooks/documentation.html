Python

import hmac
import hashlib
import time

def verify_webhook(payload: str, signature_header: str, secret: str) -> bool:
    """Verify a Spwig webhook signature."""
    # Parse the signature header
    parts = dict(p.split('=', 1) for p in signature_header.split(','))
    timestamp = parts['t']
    signature = parts['v1']

    # Check timestamp is within 5 minutes
    if abs(time.time() - int(timestamp)) > 300:
        return False

    # Compute expected signature
    expected = hmac.new(
        secret.encode(),
        f"{timestamp}.{payload}".encode(),
        hashlib.sha256
    ).hexdigest()

    # Constant-time comparison
    return hmac.compare_digest(signature, expected)


# Flask example
from flask import Flask, request, jsonify

app = Flask(__name__)
WEBHOOK_SECRET = "your_webhook_secret"

@app.route('/webhook', methods=['POST'])
def handle_webhook():
    signature = request.headers.get('X-Spwig-Signature')
    payload = request.get_data(as_text=True)

    if not verify_webhook(payload, signature, WEBHOOK_SECRET):
        return jsonify({'error': 'Invalid signature'}), 401

    data = request.json
    event_type = data['event']

    # Handle different event types
    if event_type == 'order.created':
        handle_new_order(data['data'])
    elif event_type == 'order.paid':
        handle_order_payment(data['data'])

    return jsonify({'received': True}), 200

Node.js

const crypto = require('crypto');
const express = require('express');

const WEBHOOK_SECRET = 'your_webhook_secret';

function verifyWebhook(payload, signatureHeader, secret) {
    // Parse the signature header
    const parts = Object.fromEntries(
        signatureHeader.split(',').map(p => p.split('='))
    );
    const timestamp = parts.t;
    const signature = parts.v1;

    // Check timestamp is within 5 minutes
    const now = Math.floor(Date.now() / 1000);
    if (Math.abs(now - parseInt(timestamp)) > 300) {
        return false;
    }

    // Compute expected signature
    const expected = crypto
        .createHmac('sha256', secret)
        .update(`${timestamp}.${payload}`)
        .digest('hex');

    // Constant-time comparison
    return crypto.timingSafeEqual(
        Buffer.from(signature),
        Buffer.from(expected)
    );
}

const app = express();
app.use(express.raw({ type: 'application/json' }));

app.post('/webhook', (req, res) => {
    const signature = req.headers['x-spwig-signature'];
    const payload = req.body.toString();

    if (!verifyWebhook(payload, signature, WEBHOOK_SECRET)) {
        return res.status(401).json({ error: 'Invalid signature' });
    }

    const data = JSON.parse(payload);
    const eventType = data.event;

    // Handle different event types
    switch (eventType) {
        case 'order.created':
            handleNewOrder(data.data);
            break;
        case 'order.paid':
            handleOrderPayment(data.data);
            break;
    }

    res.json({ received: true });
});

app.listen(3000);

PHP

<?php

function verifyWebhook($payload, $signatureHeader, $secret) {
    // Parse the signature header
    $parts = [];
    foreach (explode(',', $signatureHeader) as $part) {
        list($key, $value) = explode('=', $part, 2);
        $parts[$key] = $value;
    }
    $timestamp = $parts['t'];
    $signature = $parts['v1'];

    // Check timestamp is within 5 minutes
    if (abs(time() - intval($timestamp)) > 300) {
        return false;
    }

    // Compute expected signature
    $expected = hash_hmac('sha256', $timestamp . '.' . $payload, $secret);

    // Constant-time comparison
    return hash_equals($signature, $expected);
}

// Usage
$payload = file_get_contents('php://input');
$signature = $_SERVER['HTTP_X_SPWIG_SIGNATURE'] ?? '';
$secret = 'your_webhook_secret';

if (!verifyWebhook($payload, $signature, $secret)) {
    http_response_code(401);
    echo json_encode(['error' => 'Invalid signature']);
    exit;
}

$data = json_decode($payload, true);
$eventType = $data['event'];

// Handle different event types
switch ($eventType) {
    case 'order.created':
        handleNewOrder($data['data']);
        break;
    case 'order.paid':
        handleOrderPayment($data['data']);
        break;
}

echo json_encode(['received' => true]);
